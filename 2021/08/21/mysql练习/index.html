mysql练习

数据库中常使用的操作
SELECT NAME,age FROM student; – 查询姓名和年龄

SELECT address FROM student; – 查询地址

SELECT DISTINCT address FROM student; – 去除以上结果集的重复内容

SELECT NAME,math,english,math+english FROM student;– 计算math和english分数之和

– 以上结果集中柳青的和为null,因为如果有null参与运算，计算的结果都是null，使用以下操作：
SELECT NAME,math,english,math+IFNULL(english,0)FROM student;– 如果出现null用0代替

– 起别名，使用AS ,如下(当然as可以省略):
SELECT NAME,math 数学,english 英语,math + IFNULL(english,0) 总分 FROM student;

– 条件查询
SELECT * FROM student WHERE age>20 && age<30;– 查询年龄大于20小于30的人
SELECT * FROM student WHERE NAME LIKE ‘马%’;– 查询名字第一个是马的人
SELECT * FROM student WHERE NAME LIKE ‘_化%’– 查询名字第二个是化的人
– 排序查询
SELECT * FROM student ORDER BY math ASC;– 按照数学成绩升序进行排序，ASC可以不写，默认升序
SELECT * FROM student ORDER BY math ASC,english ASC;– 按照数学成绩升序进行排序,数学成绩一样就按英语成绩升序排序
– 聚合函数查询
SELECT COUNT(NAME) FROM student;– 统计有多少学生
SELECT COUNT(IFNULL(english,0)) AS 英语 FROM student;– 根据英语成绩统计学生人数
– 求出math的最大，最小，总分和平均值
SELECT MAX(math) FROM student;
SELECT MIN(math) FROM student;
SELECT SUM(math) FROM student;
SELECT AVG(math) FROM student;
– 分组查询
SELECT sex,AVG(math),COUNT(id) FROM student GROUP BY sex;– 按照性别进行分组，分别查询男女同学的数学平均分和人数(如果分组之后查询的字段不是分组字段没有任何意义(sex,sex)。)
– 按照性别进行分组，分别查询男，女同学的数学平均分和人数,要求分数低于70分的不参与分组,且分组之后只显示人数大于2个人的
SELECT sex,AVG(math),COUNT(id) FROM student WHERE math>=70 GROUP BY sex;
– 分页查询
– 每页显示三条数据
SELECT * FROM student LIMIT 0,3; – 第一页 0为开始的下标，3为显示的条数
SELECT * FROM student LIMIT 3,3; – 第二页
